// Function to fetch and render questions
async function loadQuestions(mode) {
    // 1. Hide the Home Section, Show the Assessment Section
    document.getElementById('mode-selection').classList.add('hidden'); // Ensure you have a CSS class .hidden { display: none; }
    document.getElementById('assessment-section').classList.remove('hidden');

    // 2. Fetch data from backend
    try {
        const response = await fetch(`/api/questions/${mode}`);
        const questions = await response.json();
        
        // 3. Render the HTML
        const container = document.getElementById('questions-container'); // You need a <div id="questions-container"></div> in your HTML
        container.innerHTML = ''; // Clear previous

        questions.forEach((q, index) => {
            let html = `<div class="mb-6 p-4 border rounded bg-white shadow-sm">
                          <p class="font-bold mb-2">${index + 1}. ${q.text}</p>`;
            
            if (q.type === 'mcq') {
                // Render Radio Buttons for MCQs
                q.options.forEach(opt => {
                    html += `
                    <label class="block mb-2 cursor-pointer">
                        <input type="radio" name="q_${q.id}" value="${opt}" class="mr-2">
                        ${opt}
                    </label>`;
                });
            } else if (q.type === 'text') {
                // Render Textarea for Short Answer
                html += `<textarea name="q_${q.id}" class="w-full p-2 border rounded" rows="3" placeholder="Type your answer..."></textarea>`;
            }
            
            html += `</div>`;
            container.innerHTML += html;
        });

    } catch (error) {
        alert("Error loading questions: " + error);
    }
}
